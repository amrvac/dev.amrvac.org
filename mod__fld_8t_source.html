159 !> w[iw]=w[iw]+qdt*S[wCT,qtC,x] where S is the source based on wCT within ixO
160 !> This subroutine handles the radiation force and its work added explicitly
161 !> and the energy interaction term combined with photon tiring using an implicit update
162 subroutine add_fld_rad_force(qdt,ixI^L,ixO^L,wCT,wCTprim,w,x,qsourcesplit,active)
163 use mod_constants
164 use mod_global_parameters
165 use mod_geometry
166 use mod_physics, only: phys_get_rfactor
167 integer, intent(in) :: ixi^l, ixo^l
168 double precision, intent(in) :: qdt, x(ixi^s,1:ndim)
169 double precision, intent(in) :: wct(ixi^s,1:nw),wctprim(ixi^s,1:nw)
170 double precision, intent(inout) :: w(ixi^s,1:nw)
171 logical, intent(in) :: qsourcesplit
172 logical, intent(inout) :: active
173
174 integer :: idir,jdir,nth_for_fld,ix^d
175 double precision :: sigma_b,cc
176 double precision, dimension(ixI^S) :: a1,a2,a3,c0,c1,kappa
177 double precision, dimension(ixI^S) :: e_gas,e_rad,tmp
178 double precision, dimension(ixI^S,1:ndim,1:ndim) :: div_v,edd
179
180 !> Calculate and add sourceterms
181 if(qsourcesplit .eqv. fld_radforce_split) then
182 active = .true.
183 nth_for_fld=2
184 ! store here lambda in a1 and fld_R in a2
185 call fld_get_eddington(wctprim,x,ixi^l,ixo^l,edd,a1,a2,nth_for_fld)
186 do idir = 1,ndim
187 call gradient(wctprim(ixi^s,iw_r_e),ixi^l,ixo^l,idir,tmp,nth_for_fld)
188 ! Radiation force = kappa*rho/c *Flux = lambda gradE
189 ! recycle grad E to store -lambda (grad E)_i
190 tmp(ixo^s) = -a1(ixo^s)*tmp(ixo^s)
191 !> Momentum equation source term
192 w(ixo^s,iw_mom(idir)) = w(ixo^s,iw_mom(idir))+ qdt*tmp(ixo^s)
193 !> Energy equation source term
194 w(ixo^s,iw_e) = w(ixo^s,iw_e) + qdt*wctprim(ixo^s,iw_mom(idir))*tmp(ixo^s)
195 enddo
196
197 !> Photon tiring : calculate tensor grad v (named div_v here)
198 do idir = 1,ndim
199 do jdir = 1,ndim
200 call gradient(wctprim(ixi^s,iw_mom(jdir)),ixi^l,ixo^l,idir,tmp)
201 div_v(ixo^s,idir,jdir) = tmp(ixo^s)
202 enddo
203 enddo
204 ! perform contraction fe : grad(v) with fe eddington tensor
205 {^ifoned
206 a3(ixo^s) = div_v(ixo^s,1,1)*edd(ixo^s,1,1)
207 }
208 {^iftwod
209 a3(ixo^s) = div_v(ixo^s,1,1)*edd(ixo^s,1,1) &
210 + div_v(ixo^s,1,2)*edd(ixo^s,1,2) &
211 + div_v(ixo^s,2,1)*edd(ixo^s,2,1) &
212 + div_v(ixo^s,2,2)*edd(ixo^s,2,2)
213 }
214 {^ifthreed
215 a3(ixo^s) = div_v(ixo^s,1,1)*edd(ixo^s,1,1) &
216 + div_v(ixo^s,1,2)*edd(ixo^s,1,2) &
217 + div_v(ixo^s,1,3)*edd(ixo^s,1,3) &
218 + div_v(ixo^s,2,1)*edd(ixo^s,2,1) &
219 + div_v(ixo^s,2,2)*edd(ixo^s,2,2) &
220 + div_v(ixo^s,2,3)*edd(ixo^s,2,3) &
221 + div_v(ixo^s,3,1)*edd(ixo^s,3,1) &
222 + div_v(ixo^s,3,2)*edd(ixo^s,3,2) &
223 + div_v(ixo^s,3,3)*edd(ixo^s,3,3)
224 }
225 call fld_get_opacity_prim(wctprim,x,ixi^l,ixo^l,kappa)
226
227 cc = const_c/unit_velocity
228 sigma_b = const_rad_a*cc/4.d0*(unit_temperature**4.d0)/(unit_pressure)
229 !> e_gas is the INTERNAL ENERGY without KINETIC ENERGY
230 e_gas(ixo^s) = w(ixo^s,iw_e)-half*sum(w(ixo^s,iw_mom(:))**2,dim=ndim+1)/w(ixo^s,iw_rho)
231 if(allocated(iw_mag)) then
232 e_gas(ixo^s) = e_gas(ixo^s)-half*sum(w(ixo^s,iw_mag(:))**2,dim=ndim+1)
233 endif
234 {do ix^d = ixomin^d,ixomax^d\ }
235 e_gas(ix^d) = max(e_gas(ix^d),small_pressure/(fld_gamma-1.d0))
236 {enddo\}
237 e_rad(ixo^s) = w(ixo^s,iw_r_e)
238 call phys_get_rfactor(wct,x,ixi^l,ixo^l,tmp)
239 !> Coefficients for the polynomial in Moens et al. 2022, eq 37. but with photon tiring (a3)
240 a1(ixo^s) = 4.d0*sigma_b*qdt*kappa(ixo^s)*(fld_gamma-one)**4/(wct(ixo^s,iw_rho)**3*tmp(ixo^s)**4)
241 a2(ixo^s) = cc*kappa(ixo^s)*wct(ixo^s,iw_rho)*qdt
242 a3(ixo^s) = a3(ixo^s)*qdt
243
244 c0(ixo^s) = ((one+a2(ixo^s)+a3(ixo^s))*e_gas(ixo^s)+a2(ixo^s)*e_rad(ixo^s))/a1(ixo^s)/(one+a3(ixo^s))
245 c1(ixo^s) = (one+a2(ixo^s)+a3(ixo^s))/a1(ixo^s)/(one+a3(ixo^s))
246
247 !> Loop over every cell for rootfinding method
248 {do ix^d = ixomin^d,ixomax^d\}
249 select case(fld_interaction_method)
250 case('Bisect')
251 call bisection_method(e_gas(ix^d),e_rad(ix^d),c0(ix^d),c1(ix^d))
252 case('Newton')
253 call newton_method(e_gas(ix^d),e_rad(ix^d),c0(ix^d),c1(ix^d))
254 case('Halley')
255 call halley_method(e_gas(ix^d),e_rad(ix^d),c0(ix^d),c1(ix^d))
256 case default
257 call mpistop('root-method not known')
258 end select
259 {enddo\}
260
261 e_rad(ixo^s) = (a1(ixo^s)*e_gas(ixo^s)**4.d0+e_rad(ixo^s))/(one+a2(ixo^s)+a3(ixo^s))
262
263
264 if(fix_small_values)then
265 {do ix^d = ixomin^d,ixomax^d\ }
266 e_gas(ix^d) = max(e_gas(ix^d),small_pressure/(fld_gamma-1.d0))
267 e_rad(ix^d) = max(e_rad(ix^d),small_r_e)
268 {enddo\}
269 endif
270
271 !> Update gas-energy in w, internal + kinetic
272 w(ixo^s,iw_e) = e_gas(ixo^s)
273 w(ixo^s,iw_e) = w(ixo^s,iw_e)+half*sum(w(ixo^s,iw_mom(:))**2,dim=ndim+1)/w(ixo^s,iw_rho)
274 if(allocated(iw_mag)) then
275 w(ixo^s,iw_e) = w(ixo^s,iw_e)+half*sum(w(ixo^s,iw_mag(:))**2,dim=ndim+1)
276 endif
277 !> Update rad-energy in w
278 w(ixo^s,iw_r_e) = e_rad(ixo^s)
279 end if
280 end subroutine add_fld_rad_force

282 !> get dt limit for radiation force and FLD explicit source additions
283 !> NOTE: w is primitive on entry
284 subroutine fld_radforce_get_dt(w,ixI^L,ixO^L,dtnew,dx^D,x)
285 use mod_global_parameters
286 use mod_usr_methods
287 use mod_geometry
288 use mod_physics, only: phys_get_csrad2
289
290 integer, intent(in) :: ixi^l, ixo^l
291 double precision, intent(in) :: dx^d, x(ixi^s,1:ndim), w(ixi^s,1:nw)
292 double precision, intent(inout) :: dtnew
293
294 integer :: idim,idims,nth_for_fld
295 double precision :: dxinv(1:ndim), max_grav
296 double precision :: lambda(ixi^s),fld_r(ixi^s)
297 double precision :: tmp(ixi^s)
298 double precision :: cmax(ixi^s),cmaxtot(ixi^s),courantmaxtots
299
300 nth_for_fld=2
301 call fld_get_fluxlimiter_prim(w,x,ixi^l,ixo^l,lambda,fld_r,nth_for_fld)
302 if(slab_uniform) then
303 ^d&dxinv(^d)=one/dx^d;
304 do idim = 1, ndim
305 call gradient(w(ixi^s,iw_r_e),ixi^l,ixo^l,idim,tmp,nth_for_fld)
306 max_grav = maxval(dabs(-lambda(ixo^s)*tmp(ixo^s)/w(ixo^s,iw_rho)))
307 max_grav = max(max_grav, epsilon(1.0d0))
308 dtnew = min(dtnew, 1.0d0 / dsqrt(max_grav * dxinv(idim)))
309 end do
310 else
311 do idim = 1, ndim
312 call gradient(w(ixi^s,iw_r_e),ixi^l,ixo^l,idim,tmp,nth_for_fld)
313 max_grav = maxval(dabs(-lambda(ixo^s)*tmp(ixo^s)/w(ixo^s,iw_rho))/block%ds(ixo^s,idim))
314 max_grav = max(max_grav, epsilon(1.0d0))
315 dtnew = min(dtnew, 1.0d0 / dsqrt(max_grav))
316 end do
317 endif
318
319 print *,'DT limit RADFORCE eff grav=',dtnew
320
321 ! here we interface back to fld_get_radpress
322 call phys_get_csrad2(w,x,ixi^l,ixo^l,tmp)
323 if(slab_uniform) then
324 ^d&dxinv(^d)=one/dx^d;
325 do idims=1,ndim
326 cmax(ixo^s)=dabs(w(ixo^s,iw_mom(idims)))+dsqrt(tmp(ixo^s))
327 if(idims==1) then
328 cmaxtot(ixo^s)=cmax(ixo^s)*dxinv(idims)
329 else
330 cmaxtot(ixo^s)=cmaxtot(ixo^s)+cmax(ixo^s)*dxinv(idims)
331 end if
332 end do
333 else
334 do idims=1,ndim
335 cmax(ixo^s)=dabs(w(ixo^s,iw_mom(idims)))+dsqrt(tmp(ixo^s))
336 if(idims==1) then
337 cmaxtot(ixo^s)=cmax(ixo^s)/block%ds(ixo^s,idims)
338 else
339 cmaxtot(ixo^s)=cmaxtot(ixo^s)+cmax(ixo^s)/block%ds(ixo^s,idims)
340 end if
341 end do
342 end if
343 ! courantmaxtots='max(summed c/dx)'
344 courantmaxtots=maxval(cmaxtot(ixo^s))
345 if(courantmaxtots>smalldouble) dtnew=min(dtnew,courantpar/courantmaxtots)
346 print *,'DT limit RADFORCE CSRAD=',dtnew
347
348 end subroutine fld_radforce_get_dt

441 !> This subroutine calculates flux limiter lambda according to fld_fluxlimiter
442 !> It also calculates fld_R which is ratio of radiation scaleheight and mean free path
443 !> NOTE: this one operates on primitives
444 !> NOTE: nth and ixI and ixO not free to choose
445 subroutine fld_get_fluxlimiter_prim(w,x,ixI^L,ixO^L,fld_lambda,fld_R,nth)
446 use mod_global_parameters
447 use mod_geometry
448 use mod_usr_methods
449 integer, intent(in) :: ixi^l,ixo^l,nth
450 double precision, intent(in) :: w(ixi^s,1:nw)
451 double precision, intent(in) :: x(ixi^s,1:ndim)
452 double precision, intent(out) :: fld_r(ixi^s),fld_lambda(ixi^s)
453
454 integer :: idir, ix^d
455 double precision :: kappa(ixi^s),normgrad2(ixi^s)
456 double precision :: grad_r_e(ixi^s)
457
458 select case(fld_fluxlimiter)
459 case('Diffusion')
460 ! optically thick limit
461 fld_lambda(ixo^s) = 1.d0/3.d0
462 fld_r(ixo^s) = zero
463 case('FreeStream')
464 ! optically thin limit
465 normgrad2(ixo^s) = zero
466 do idir=1,ndim
467 call gradient(w(ixi^s,iw_r_e),ixi^l,ixo^l,idir,grad_r_e,nth)
468 normgrad2(ixo^s) = normgrad2(ixo^s)+grad_r_e(ixo^s)**2
469 end do
470 call fld_get_opacity_prim(w,x,ixi^l,ixo^l,kappa)
471 ! Calculate R everywhere
472 ! |grad E|/(rho kappa E)
473 fld_r(ixo^s) = dsqrt(normgrad2(ixo^s))/(kappa(ixo^s)*w(ixo^s,iw_rho)*w(ixo^s,iw_r_e))
474 where(normgrad2(ixo^s)<smalldouble**2)
475 ! treat uniform case as diffusion limit
476 fld_r(ixo^s)=zero
477 fld_lambda(ixo^s) = 1.0d0/3.0d0
478 elsewhere
479 fld_lambda(ixo^s) = one/fld_r(ixo^s)
480 endwhere
481 case('Pomraning')
482 ! Calculate R everywhere
483 ! |grad E|/(rho kappa E)
484 normgrad2(ixo^s) = zero
485 do idir = 1,ndim
486 call gradient(w(ixi^s,iw_r_e),ixi^l,ixo^l,idir,grad_r_e,nth)
487 normgrad2(ixo^s) = normgrad2(ixo^s) + grad_r_e(ixo^s)**2
488 end do
489 call fld_get_opacity_prim(w,x,ixi^l,ixo^l,kappa)
490 fld_r(ixo^s) = dsqrt(normgrad2(ixo^s))/(kappa(ixo^s)*w(ixo^s,iw_rho)*w(ixo^s,iw_r_e))
491 ! Calculate the flux limiter, lambda
492 ! Levermore and Pomraning: lambda = (2 + R)/(6 + 3R + R^2)
493 fld_lambda(ixo^s) = (2.d0+fld_r(ixo^s))/(6.d0+3*fld_r(ixo^s)+fld_r(ixo^s)**2)
494 case('Minerbo')
495 ! Calculate R everywhere
496 ! |grad E|/(rho kappa E)
497 normgrad2(ixo^s) = zero
498 do idir = 1,ndim
499 call gradient(w(ixi^s,iw_r_e),ixi^l,ixo^l,idir,grad_r_e,nth)
500 normgrad2(ixo^s) = normgrad2(ixo^s) + grad_r_e(ixo^s)**2
501 end do
502 call fld_get_opacity_prim(w, x, ixi^l, ixo^l, kappa)
503 fld_r(ixo^s) = dsqrt(normgrad2(ixo^s))/(kappa(ixo^s)*w(ixo^s,iw_rho)*w(ixo^s,iw_r_e))
504 ! Calculate the flux limiter, lambda
505 ! Minerbo:
506 {do ix^d = ixomin^d,ixomax^d\ }
507 if(fld_r(ix^d) .lt. 3.d0/2.d0) then
508 fld_lambda(ix^d) = 2.d0/(3.d0+dsqrt(9.d0+12.d0*fld_r(ix^d)**2))
509 else
510 fld_lambda(ix^d) = 1.d0/(1.d0+fld_r(ix^d)+dsqrt(1.d0+2.d0*fld_r(ix^d)))
511 endif
512 {enddo\}
513 case('special')
514 if (.not. associated(usr_special_fluxlimiter)) then
515 call mpistop("special fluxlimiter not defined")
516 endif
517 call usr_special_fluxlimiter(ixi^l,ixo^l,w,x,fld_lambda,fld_r)
518 case default
519 call mpistop('Fluxlimiter unknown')
520 end select
521
522 end subroutine fld_get_fluxlimiter_prim

558 !> Calculate Eddington-tensor (where w is primitive)
559 !> also feeds back the flux limiter lambda and R
560 subroutine fld_get_eddington(w, x, ixI^L, ixO^L, eddington_tensor, lambda, fld_R, nth)
561 use mod_global_parameters
562 use mod_geometry
563 integer, intent(in) :: ixI^L, ixO^L, nth
564 double precision, intent(in) :: w(ixI^S, 1:nw)
565 double precision, intent(in) :: x(ixI^S, 1:ndim)
566 double precision, intent(out) :: eddington_tensor(ixI^S,1:ndim,1:ndim)
567 double precision, intent(out) :: lambda(ixI^S),fld_R(ixI^S)
568
569 integer :: idir,jdir
570 double precision :: normgrad2(ixI^S)
571 double precision :: tmp(ixI^S),grad_r_e(ixI^S,1:ndim)
572 double precision :: nn_regularized(ixI^S,1:ndim,1:ndim)
573
574 normgrad2(ixo^s) = zero
575 do idir = 1,ndim
576 call gradient(w(ixi^s, iw_r_e),ixi^l,ixo^l,idir,tmp,nth)
577 grad_r_e(ixo^s,idir)=tmp(ixo^s)
578 normgrad2(ixo^s)=normgrad2(ixo^s)+tmp(ixo^s)**2
579 end do
580 do idir = 1,ndim
581 do jdir = 1,ndim
582 if(idir==jdir)then
583 nn_regularized(ixo^s,idir,jdir)=(grad_r_e(ixo^s,idir)*grad_r_e(ixo^s,jdir)+smalldouble**2)/(normgrad2(ixo^s)+smalldouble**2)
584 else
585 nn_regularized(ixo^s,idir,jdir)=(grad_r_e(ixo^s,idir)*grad_r_e(ixo^s,jdir))/(normgrad2(ixo^s)+smalldouble**2)
586 endif
587 enddo
588 enddo
589 ! get lambda and R
590 call fld_get_fluxlimiter_prim(w,x,ixi^l,ixo^l,lambda,fld_r,nth)
591 ! store f_e= lambda + lambda^2 R^2
592 tmp(ixo^s) = lambda(ixo^s)+(lambda(ixo^s)*fld_r(ixo^s))**2
593 do idir = 1,ndim
594 ! first compute the isotropic (diagonal) part
595 eddington_tensor(ixo^s,idir,idir) = half*(one-tmp(ixo^s))
596 enddo
597 do idir = 1,ndim
598 do jdir = 1,ndim
599 ! initialize off-diagonal part here
600 if(idir .ne. jdir) eddington_tensor(ixo^s,idir,jdir) = zero
601 ! add part depending on unit vectors along gradient E
602 eddington_tensor(ixo^s,idir,jdir) = eddington_tensor(ixo^s,idir,jdir)+&
603 half*(3.d0*tmp(ixo^s)-one)*nn_regularized(ixo^s,idir,jdir)
604 enddo
605 enddo
606 end subroutine fld_get_eddington

608 !> Calling all subroutines to perform the multigrid method
609 !> Communicates rad_e and diff_coeff to multigrid library
610 !> Advance psa=psb+dtfactor*qdt*F_im(psa)
611 subroutine fld_implicit_update(dtfactor,qdt,qtC,psa,psb)
612 use mod_global_parameters
613 use mod_forest
614 use mod_multigrid_coupling
615 type(state), target :: psa(max_blocks)
616 type(state), target :: psb(max_blocks)
617 double precision, intent(in) :: qdt
618 double precision, intent(in) :: qtC
619 double precision, intent(in) :: dtfactor
620
621 integer, parameter :: max_its = 100
622 integer :: n,ixO^L,ix^D
623 double precision :: res, max_residual, lambda, fac
624 integer :: iigrid, igrid
625
626 ! Avoid setting a very restrictive limit to residual when time step small
627 if(qdt < dtmin) then
628 if(mype==0)then
629 print *,'skipping implicit solve: dt too small!'
630 print *,'Currently at time=',global_time,' time step=',qdt,' dtmin=',dtmin
631 endif
632 return
633 endif
634 max_residual = fld_diff_tol
635
636 call mg_set_methods(mg)
637 if(.not. mg%is_allocated) call mpistop("multigrid tree not allocated yet")
638
639 ! Here we handle the global helmholtz problem with variable coefficient
640 ! The equation we solve is div(D nabla Erad^(n+1)) -(1/dt)Erad^(n+1)=-(1/dt)Erad^n
641 ! Helmholtz equation is div(eps nabla phi) -lambda phi = f
642 ! hence eps is our variable coefficient D=fld_lambda*c/(kappa*rho)
643 ! hence phi is Erad and lambda=1/dt
644 lambda = 1.d0/(dtfactor *qdt)
645 fac = 1.d0
646 call vhelmholtz_set_lambda(lambda)
647 call update_diffcoeff(psa)
648 ! copy in the (variable) diffusion coefficient in mg_iveps
649 call mg_copy_to_tree(i_diff_mg, mg_iveps, factor=fac, state_from=psa)
650 ! copy in the Erad variable in mg_iphi (the one we solve for)
651 call mg_copy_to_tree(iw_r_e, mg_iphi, factor=fac, state_from=psa)
652 ! copy in RHS f factor as Erad with factor -(1/dt)
653 call mg_copy_to_tree(iw_r_e, mg_irhs, factor=-lambda, state_from=psb)
654 if(time_advance)then
655 call mg_restrict(mg, mg_iveps)
656 call mg_fill_ghost_cells(mg, mg_iveps)
657 endif
658 call mg_fas_fmg(mg, .true., max_res=res)
659 do n = 1, max_its
660 if(res < max_residual) exit
661 call mg_fas_vcycle(mg, max_res=res)
662 end do
663 if((res .le. 0.d0) .or. (res .ge. max_residual)) then
664 if (mg%my_rank == 0) then
665 write(*,*) it, ' residual from MG ', res
666 write(*,*) it, ' max_residual in MG ', max_residual
667 write(*,*) it, ' qdt in MG ', qdt, ' versus dtmin=',dtmin
668 write(*,*) it, ' dtfactor*qdt in MG ', qdt*dtfactor
669 endif
670 call mpistop("no convergence in MG")
671 end if
672 ! copy back the Erad variable in iw_r_e
673 call mg_copy_from_tree_gc(mg_iphi, iw_r_e, state_to=psa)
674
675 if(fix_small_values)then
676 ixo^l=ixm^ll;
677 do iigrid=1,igridstail; igrid=igrids(iigrid);
678 {do ix^d = ixomin^d,ixomax^d\ }
679 psa(igrid)%w(ix^d,iw_r_e)= max(psa(igrid)%w(ix^d,iw_r_e),small_r_e)
680 {enddo\}
681 end do
682 endif
683
684 end subroutine fld_implicit_update

798 !> Find the root of the 4th degree polynomial using the bisection method
799 subroutine bisection_method(e_gas, E_rad, c0, c1)
800 use mod_global_parameters
801 double precision, intent(in) :: c0, c1
802 double precision, intent(in) :: E_rad
803 double precision, intent(inout) :: e_gas
804 double precision :: bisect_a, bisect_b, bisect_c
805 integer :: n, max_its
806
807 n = 0
808 max_its = 100
809 bisect_a = zero
810 bisect_b = min(dabs(c0/c1),dabs(c0)**(1.d0/4.d0))+smalldouble
811 do while(dabs(bisect_b-bisect_a) .ge. fld_bisect_tol*min(e_gas,e_rad))
812 bisect_c = (bisect_a + bisect_b)/two
813 n = n +1
814 if(n .gt. max_its) then
815 call mpistop('No convergece in bisection scheme')
816 endif
817 if(polynomial_bisection(bisect_a, c0, c1)*&
818 polynomial_bisection(bisect_b, c0, c1) .lt. zero) then
819 if(polynomial_bisection(bisect_a, c0, c1)*&
820 polynomial_bisection(bisect_c, c0, c1) .lt. zero) then
821 bisect_b = bisect_c
822 elseif(polynomial_bisection(bisect_b, c0, c1)*&
823 polynomial_bisection(bisect_c, c0, c1) .lt. zero) then
824 bisect_a = bisect_c
825 elseif(polynomial_bisection(bisect_a, c0, c1) .eq. zero) then
826 bisect_b = bisect_a
827 bisect_c = bisect_a
828 goto 2435
829 elseif(polynomial_bisection(bisect_b, c0, c1) .eq. zero) then
830 bisect_a = bisect_b
831 bisect_c = bisect_b
832 goto 2435
833 elseif(polynomial_bisection(bisect_c, c0, c1) .eq. zero) then
834 bisect_a = bisect_c
835 bisect_b = bisect_c
836 goto 2435
837 else
838 call mpistop("Problem with fld bisection method")
839 endif
840 elseif(polynomial_bisection(bisect_a, c0, c1) &
841 - polynomial_bisection(bisect_b, c0, c1) .lt. fld_bisect_tol*polynomial_bisection(bisect_a, c0, c1)) then
842 goto 2435
843 else
844 bisect_a = e_gas
845 bisect_b = e_gas
846 print*, "IGNORING GAS-RAD ENERGY EXCHANGE ", c0, c1
847 print*, polynomial_bisection(bisect_a, c0, c1), polynomial_bisection(bisect_b, c0, c1)
848 call mpistop('issues in bisection scheme')
849 if(polynomial_bisection(bisect_a, c0, c1) .le. smalldouble) then
850 bisect_b = bisect_a
851 elseif(polynomial_bisection(bisect_a, c0, c1) .le. smalldouble) then
852 bisect_a = bisect_b
853 endif
854 goto 2435
855 endif
856 enddo
857 2435 e_gas = (bisect_a + bisect_b)/two
858 end subroutine bisection_method

860 !> Find the root of the 4th degree polynomial using the Newton method
861 subroutine newton_method(e_gas, E_rad, c0, c1)
862 use mod_global_parameters
863 double precision, intent(in) :: c0, c1
864 double precision, intent(in) :: E_rad
865 double precision, intent(inout) :: e_gas
866 double precision :: xval, yval, der, deltax
867 integer :: ii
868
869 yval = bigdouble
870 xval = e_gas
871 der = one
872 deltax = one
873 ii = 0
874 !> Compare error with dx = dx/dy dy
875 do while(dabs(deltax) .gt. fld_bisect_tol)
876 yval = polynomial_bisection(xval, c0, c1)
877 der = dpolynomial_bisection(xval, c0, c1)
878 deltax = -yval/der
879 xval = xval + deltax
880 ii = ii + 1
881 if(ii .gt. 1d3) then
882 print*, 'skip to bisection algorithm'
883 call bisection_method(e_gas, e_rad, c0, c1)
884 return
885 endif
886 enddo
887 e_gas = xval
888 end subroutine newton_method

890 !> Find the root of the 4th degree polynomial using the Halley method
891 subroutine halley_method(e_gas, E_rad, c0, c1)
892 use mod_global_parameters
893 double precision, intent(in) :: c0, c1
894 double precision, intent(in) :: E_rad
895 double precision, intent(inout) :: e_gas
896 double precision :: xval, yval, der, dder, deltax
897 integer :: ii
898
899 yval = bigdouble
900 xval = e_gas
901 der = one
902 dder = one
903 deltax = one
904 ii = 0
905 !> Compare error with dx = dx/dy dy

914 print*, 'skip to Newton algorithm'
915 !call mpistop("iterates to 1000, entering Newton method")

918 endif
919 enddo
920 e_gas = xval
921 end subroutine halley_method

943 !> Evaluate second derivative of polynomial at argument e_gas
944 function ddpolynomial_bisection(e_gas, c0, c1) result(dder)
945 use mod_global_parameters
946 double precision, intent(in) :: e_gas
