1195 subroutine trace_qperp_compute_scalars(result,field_min)

1202 bf_norm=dsqrt(sum(result%forward_B**2))

1204 if (bseed_norm<=zero .or. bf_norm<=zero .or. bb_norm<=zero .or. &
1205 bseed_norm<field_min .or. bf_norm<field_min .or. &
1206 bb_norm<field_min) then

1212 dot_b=sum(result%u_backward_perp*result%v_backward_perp)
1213 result%N2=sum(result%u_forward_perp**2)* &
1214 sum(result%v_backward_perp**2)+ &
1215 sum(result%u_backward_perp**2)* &
1216 sum(result%v_forward_perp**2)-2.d0*dot_f*dot_b
1217 result%bfactor=abs(bf_norm*bb_norm)/(bseed_norm**2)
